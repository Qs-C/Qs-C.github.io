平衡树

//数据结构初始化
struct node{
    int val,rnk,lc,rc,size;//值，优先级，左右子节点，子树大小
}tree[N];
int tot,root;//总结点数和根节点数

//节点管理
void update(int k){
    tree[k].size=tree[tree[k].lc].size+tree[tree[k].rc].size+1;
    return;
}
int add_node(int val){
    tree[++tot].size=1;
    tree[tot].val=0;
    tree[tot].lc=tree[tot].rc=0;
    tree[tot].rnk=rand();
    return tot;
}

void split(int k,int &a,int &b,int val){
    if(k==0){
        a=b=0;
        return ;
    }
    if(tree[k].val<=val){
        a=k;
        split(tree[k].rc,tree[k].rc,b,val);
    }
    else {
        b=k;
        split(tree[k].lc,a,tree[k].lc,val);
    }
    update(k);
}
//给定一棵平衡树 ，按照某个权值val，分裂成平衡树a和b，且a树所有点的权值不超过val，b树的所有点权大于val。

void merge(int &k ,int a,int b){
    if(a==0||b==0){
        k=a+b;
        return;
    }
    if(tree[a].rnk<tree[b].rnk){
        k=a;
        merge(tree[a].rc,tree[a].rc,b);
    }
    else {
        k=b;
        merge(tree[b].lc,a,tree[b].lc);
    }
    update(k);
}

void insert(int &k ,int val){
    int a=0,b=0,cur=add_node(val);
    split(k,a,b,val);
    merge(a,a,cur);
    merge(k,a,b);
    return;
}
void del(int &k,int val){
    int a=0,b=0,z=0;
    split(k,a,b,val);
    split(a,a,z,val-1);
    merge(z,tree[z].lc,tree[z].rc);
    merge(a,a,z);
    merge(k,a,b);
    return;
}

int find_num(int k,int x){
    while(tree[tree[k].lc]+1!=x){
        if(tree[tree[k].lc].size>=x){
            k=tree[k].lc;
        }
        else {
            x-=tree[tree[k].lc].size;
            k=tree[k].rc;
        }
    }
    return tree[k].val;
}
//找到树上第x小的数

int find_rank(int &k,int val){
    int a=0,b=0;
    split(k,a,b,val-1);
    int temp=tree[a].size+1;
    merge(k,a,b);
    return;
}
//查询val的排名

int prev(int &k,int val){
    int a=0,b=0;
    split(k,a,b,val-1);
    int temp=find_num(a,tree[a].size);
    merge(k,a,b);
    return temp;
}
//查询比val小的最大值

int suf(int &k,int val){
    int a=0,b=0;
    split(k,a,b,val);
    int temp=find_num(b,1);
    merge(k,a,b);
    return temp;
}

文艺平衡树

支持区间操作的平衡树。如区间反转，区间求和/最值，区间赋值

//初始化
struct node{
    int size,lc,rc,rnk,val;
    bool rev;
}tree[N];

void pushdown(int k){
    if(tree[k].rev){
        swap(tree[k].lc,tree[k].rc);
        tree[tree[k].lc].rev^=1;
        tree[tree[k].rc].rev^=1;
        tree[k].rev=0;
    }
    return ;
}
//lazy tag方式延迟执行

void split_sz(int k,int &a,int &b,int sz){
    if(k==0){
        a=b=0;
        return;
    }
    pushdown(k);
    if(tree[tree[k].lc].size+1<=sz){
        a=k;
        split(tree[k].rc,tree[k].rc,b,sz-tree[tree[k].rc].size-1);
    }
    else {
        n=k;
        split(tree[k].lc,a,tree[k].lc,sz);
    }
}
//分出树的前sz个节点

void reverse(int l,int r){
    int a=0,b=0,c=0;
    split(root,a,b,l-1);
    split(b,b,c,r-l+1);
    tree[b].rev^=1;
    merge(a,a,b);
    merge(root,a,c);
}
//将l到r的区间反转
