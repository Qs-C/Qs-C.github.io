树形dp

在树上进行的dp，一般自下向上递归进行
在树形dp当中，我们一般先算子树再进行合并，在实现上与树的后序遍历相似，都是先遍历子树，遍历完之后将子树的值传给父亲。简单来说我们动态规划的过程大概就是先递归访问所有子树，再在根上合并

prim

选出离当前生成树最近的点加入生成树

const int max=10000,INF=0x3f3f3f3f;
int g[max][max],dist[max],n,m,res;
//g[][]保存边的权重，dist保存点到当前生成树的距离，res保存结果
bool book[max];
//book保存当前点的使用情况

//首先初始化g[][]，dist[]为INF，记录存在边的权重

void prim(){
    dist[1]=0;
    book[1]=true;
    for(int i=2;i<=n;i++)dist[i]=min(dist[i],g[1][i]);
    
    for(int i=2;i<=n;i++){
        int temp=INF;
        int t=-1;
        for(int j=2;j<=n;j++){
            if(!book[j]&&dist[j]<temp){
                t=j;
                temp=dist[j];
            }
        }
        if(t==-1){
            res=-1;
            return;
        }
        book[t]=true;
        res+=dist[t];
        for(int j=2;j<=n;j++)dist[j]=min(dist[j],g[j][t]);
    }
}

平衡树

//数据结构初始化
struct node{
    int val,rnk,lc,rc,size;//值，优先级，左右子节点，子树大小
}tree[N];
int tot,root;//总结点数和根节点数

//节点管理
void update(int k){
    tree[k].size=tree[tree[k].lc].size+tree[tree[k].rc].size+1;
    return;
}
int add_node(int val){
    tree[++tot].size=1;
    tree[tot].val=0;
    tree[tot].lc=tree[tot].rc=0;
    tree[tot].rnk=rand();
    return tot;
}

void split(int k,int &a,int &b,int val){
    if(k==0){
        a=b=0;
        return ;
    }
    if(tree[k].val<=val){
        a=k;
        split(tree[k].rc,tree[k].rc,b,val);
    }
    else {
        b=k;
        split(tree[k].lc,a,tree[k].lc,val);
    }
    update(k);
}
//给定一棵平衡树 ，按照某个权值val，分裂成平衡树a和b，且a树所有点的权值不超过val，b树的所有点权大于val。

void merge(int &k ,int a,int b){
    if(a==0||b==0){
        k=a+b;
        return;
    }
    if(tree[a].rnk<tree[b].rnk){
        k=a;
        merge(tree[a].rc,tree[a].rc,b);
    }
    else {
        k=b;
        merge(tree[b].lc,a,tree[b].lc);
    }
    update(k);
}

void insert(int &k ,int val){
    int a=0,b=0,cur=add_node(val);
    split(k,a,b,val);
    merge(a,a,cur);
    merge(k,a,b);
    return;
}
void del(int &k,int val){
    int a=0,b=0,z=0;
    split(k,a,b,val);
    split(a,a,z,val-1);
    merge(z,tree[z].lc,tree[z].rc);
    merge(a,a,z);
    merge(k,a,b);
    return;
}

int find_num(int k,int x){
    while(tree[tree[k].lc]+1!=x){
        if(tree[tree[k].lc].size>=x){
            k=tree[k].lc;
        }
        else {
            x-=tree[tree[k].lc].size;
            k=tree[k].rc;
        }
    }
    return tree[k].val;
}
//找到树上第x小的数

int find_rank(int &k,int val){
    int a=0,b=0;
    split(k,a,b,val-1);
    int temp=tree[a].size+1;
    merge(k,a,b);
    return;
}
//查询val的排名

int prev(int &k,int val){
    int a=0,b=0;
    split(k,a,b,val-1);
    int temp=find_num(a,tree[a].size);
    merge(k,a,b);
    return temp;
}
//查询比val小的最大值

int suf(int &k,int val){
    int a=0,b=0;
    split(k,a,b,val);
    int temp=find_num(b,1);
    merge(k,a,b);
    return temp;
}

heap初探 ​ ​

核心概念 ​ ​ ​

  • ​UAF（Use After Free） ​
    当一块内存（chunk）被free回堆管理器器，里面的指针没有被置空，就可以通过这个指针读出或写入内容
    当内存被复用时，可以利用悬空指针读取数据
    当内存空闲时，可以利用悬空指针修改向前指针，劫持分配流 ​ ​ ​
  • ​Tcache ​
    当释放一个堆块时,会优先放入Tcache中
    当修改空闲块的前8个字节时，就能将写入下一次申请内存时分配的地址
  • Unsorted Bin
    当Tcache被填满时，后续释放的堆块会进入Unsorted Bin
    进入这里的第一个堆块，向前和向后指针会指向libc内部地址，进而算出libc基址
  • Heap Overflow
    往堆块中写入数据时超过了申请的大小，覆盖了下一个堆块的header，就可以修改下一个堆块的大小导致堆重叠，或者将下一个堆块的P位改为0，就可以合并chunk
  • Double Free
    释放两次同一个指针，实现A->B->A，申请A时，修改内容，B的向前指针也会被修改，实现任意地址分配。
